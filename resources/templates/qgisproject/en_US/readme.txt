To update this template you must add these variables instead of the actual values saved by QGIS:
- __SRSID__ (<srsid>__SRSID__</srsid>)
- __SRID__ (<srid>__SRID__</srid>)
- __AUTHID__ (<authid>__AUTHID__</authid>)
- __XMAX__ (<xmax>__XMAX__</xmax>)
- __XMIN__ (<xmin>__XMIN__</xmin>)
- __YMAX__ (<ymax>__YMAX__</ymax>)
- __YMIN__ (<ymin>__YMIN__</ymin>)
- __DATASOURCE__ (<datasource>__DATASOURCE__|layername=[...]</datasource>)
- __GPKGPATH__ (    <Variables>
					  <variableNames type="QStringList">
						<value>project_gpkg_path</value>
					  </variableNames>
					  <variableValues type="QStringList">
						<value>__GPKGPATH__</value>
					  </variableValues>
					</Variables>
				)
